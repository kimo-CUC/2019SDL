### 实验代码

```c
#define CRT_SECLRE_NO_WARNINGS//定义宏，删掉的话不能通过
#include<stdlib.h>
#include<stdio.h>
#include<string.h>
int sub(char* x)
{
    char y[10];
    strcpy(y,x);//安全性问题
    return 0;
}
int main(int argc,char** argv)
{
    if(argc>1)
        sub(argv[1]);
    printf("exit");
}
```
